61void CViewFrmScroll::slotMouseMoveEvent(QMouseEvent *event)
62{
63 int area = 10;
64 CFrmViewer* pFrmView = qobject_cast<CFrmViewer*>(sender());
65 QScrollBar* pHBar = horizontalScrollBar();
66 QScrollBar* pVBar = verticalScrollBar();
67 if(pHBar->isHidden() && pVBar->isHidden())
68 return;
69
70 int nVWidth = 0;
71 if(!pVBar->isHidden()) nVWidth = pVBar->geometry().width();
72 QPoint pos = pFrmView->mapToParent(event->pos());
73 if(pos.x() < area && pHBar->value() > pHBar->minimum())
74 pHBar->setValue(pHBar->value() - pHBar->singleStep());
75 else if(pos.x() > width() - area - nVWidth
76 && pHBar->value() < pHBar->maximum())
77 pHBar->setValue(pHBar->value() + pHBar->singleStep());
78
79 int nHHeight = 0;
80 if(!pHBar->isHidden()) nHHeight = pHBar->geometry().height();
81 if(pos.y() < area && pVBar->value() > pVBar->minimum())
82 pVBar->setValue(pVBar->value() - pVBar->singleStep());
83 else if(pos.y() > height() - area - nHHeight
84 && pVBar->value() < pVBar->maximum())
85 pVBar->setValue(pVBar->value() + pVBar->singleStep());
86}
