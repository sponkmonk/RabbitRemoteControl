79int CConnect::SetViewer(CFrmViewer *pView, bool bDirectConnection)
80{
81 Q_ASSERT(pView);
82 if(!pView) return -1;
83
84 if(m_pView)
85 {
86 this->disconnect(m_pView);
87 }
88
89 m_pView = pView;
90 bool check = connect(this, SIGNAL(sigSetDesktopSize(int, int)),
91 m_pView, SLOT(slotSetDesktopSize(int, int)));
92 Q_ASSERT(check);
93 check = connect(this, SIGNAL(sigServerName(const QString&)),
94 m_pView, SLOT(slotSetName(const QString&)));
95 Q_ASSERT(check);
96
97 check = connect(this, SIGNAL(sigUpdateRect(const QRect&, const QImage&)),
98 m_pView, SLOT(slotUpdateRect(const QRect&, const QImage&)));
99 Q_ASSERT(check);
100 check = connect(this, SIGNAL(sigUpdateCursor(const QCursor&)),
101 m_pView, SLOT(slotUpdateCursor(const QCursor&)));
102 Q_ASSERT(check);
103 check = connect(this, SIGNAL(sigUpdateCursorPosition(const QPoint&)),
104 m_pView, SLOT(slotUpdateCursorPosition(const QPoint&)));
105 Q_ASSERT(check);
106
107 if(bDirectConnection)
108 {
109 /* \~chinese 因为连接可能是在另一个线程中的非Qt事件处理，
110 * 它可能会阻塞线程，那会导致键盘、鼠标事件延迟，
111 * 所以这里用 Qt::DirectConnection
112 * \~english Because the connection may be a non-Qt event processing in another thread,
113 * it may block the thread, which will cause the keyboard and mouse events to be delayed.
114 * So here use Qt::DirectConnection
115 */
116 check = connect(m_pView, SIGNAL(sigMousePressEvent(Qt::MouseButtons, QPoint)),
117 this, SLOT(slotMousePressEvent(Qt::MouseButtons, QPoint)),
118 Qt::DirectConnection);
119 Q_ASSERT(check);
120 check = connect(m_pView, SIGNAL(sigMouseReleaseEvent(Qt::MouseButton, QPoint)),
121 this, SLOT(slotMouseReleaseEvent(Qt::MouseButton, QPoint)),
122 Qt::DirectConnection);
123 Q_ASSERT(check);
124 check = connect(m_pView, SIGNAL(sigMouseMoveEvent(Qt::MouseButtons, QPoint)),
125 this, SLOT(slotMouseMoveEvent(Qt::MouseButtons, QPoint)),
126 Qt::DirectConnection);
127 Q_ASSERT(check);
128 check = connect(m_pView, SIGNAL(sigWheelEvent(Qt::MouseButtons, QPoint, QPoint)),
129 this, SLOT(slotWheelEvent(Qt::MouseButtons, QPoint, QPoint)),
130 Qt::DirectConnection);
131 Q_ASSERT(check);
132 check = connect(m_pView, SIGNAL(sigKeyPressEvent(int, Qt::KeyboardModifiers)),
133 this, SLOT(slotKeyPressEvent(int, Qt::KeyboardModifiers)),
134 Qt::DirectConnection);
135 Q_ASSERT(check);
136 check = connect(m_pView, SIGNAL(sigKeyReleaseEvent(int, Qt::KeyboardModifiers)),
137 this, SLOT(slotKeyReleaseEvent(int, Qt::KeyboardModifiers)),
138 Qt::DirectConnection);
139 Q_ASSERT(check);
140 } else {
141 check = connect(m_pView, SIGNAL(sigMousePressEvent(Qt::MouseButtons, QPoint)),
142 this, SLOT(slotMousePressEvent(Qt::MouseButtons, QPoint)));
143 Q_ASSERT(check);
144 check = connect(m_pView, SIGNAL(sigMouseReleaseEvent(Qt::MouseButton, QPoint)),
145 this, SLOT(slotMouseReleaseEvent(Qt::MouseButton, QPoint)));
146 Q_ASSERT(check);
147 check = connect(m_pView, SIGNAL(sigMouseMoveEvent(Qt::MouseButtons, QPoint)),
148 this, SLOT(slotMouseMoveEvent(Qt::MouseButtons, QPoint)));
149 Q_ASSERT(check);
150 check = connect(m_pView, SIGNAL(sigWheelEvent(Qt::MouseButtons, QPoint, QPoint)),
151 this, SLOT(slotWheelEvent(Qt::MouseButtons, QPoint, QPoint)));
152 Q_ASSERT(check);
153 check = connect(m_pView, SIGNAL(sigKeyPressEvent(int, Qt::KeyboardModifiers)),
154 this, SLOT(slotKeyPressEvent(int, Qt::KeyboardModifiers)));
155 Q_ASSERT(check);
156 check = connect(m_pView, SIGNAL(sigKeyReleaseEvent(int, Qt::KeyboardModifiers)),
157 this, SLOT(slotKeyReleaseEvent(int, Qt::KeyboardModifiers)));
158 Q_ASSERT(check);
159 }
160
161 return 0;
162}

185void CConnect::slotTimeOut()
186{
187 try {
188 // >= 0 : continue
189 // < 0: error or stop
190 int nTime = OnProcess();
191 if(nTime >= 0)
192 {
193 QTimer::singleShot(nTime, this, SLOT(slotTimeOut()));
194 return;
195 }
196 qCritical(Client) << "Process fail:" << nTime;
197 } catch(std::exception e) {
198 qCritical(Client) << "Process fail:" << e.what();
199 emit sigError(-1, e.what());
200 } catch (...) {
201 qCritical(Client) << "Process fail";
202 emit sigError(-2, "");
203 }
204
205 // Error or stop, must notify user disconnect it
206 emit sigDisconnected();
207}
