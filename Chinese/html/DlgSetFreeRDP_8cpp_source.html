128 m_pSettings->SetReconnectInterval(ui->sbReconnect->value());
129
130 // Redirection
131 m_pSettings->SetRedirectionPrinter(ui->cbPrinter->isChecked());
132 CParameterFreeRDP::RedirecionSoundType tRdirectionSound;

140 m_pSettings->SetRedirectionSoundParamters(ui->leRdpSnd->text());

265int CDlgSetFreeRDP::InsertDesktopSize(int width, int height)
266{
267 QString curSize = QString::number(width) + "×" + QString::number(height);
268 if(ui->cbDesktopSize->findText(curSize) > -1)
269 return 0;
270
271 for(int i = 0; i < ui->cbDesktopSize->count(); i++)
272 {
273 QString curText = ui->cbDesktopSize->itemText(i);
274 int nIndex = curText.indexOf("×");
275 if(-1 < nIndex)
276 {
277 int w = curText.left(nIndex).toInt();
278 if(w > width)
279 {
280 ui->cbDesktopSize->insertItem(i, curSize);
281 return 0;
282 }
283 }
284 }
285
286 if(ui->cbDesktopSize->findText(curSize) == -1)
287 ui->cbDesktopSize->addItem(curSize);
288
289 return 0;
290}
