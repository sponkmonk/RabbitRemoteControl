120int CFrmFullScreenToolBar::ReToolBarSize()
121{
122 int marginW = style()->pixelMetric(
123 QStyle::PM_FocusFrameHMargin) << 1;
124 int marginH = style()->pixelMetric(
125 QStyle::PM_FocusFrameVMargin) << 1;
126 QMargins cm = contentsMargins();
127
128 QSize size = m_ToolBar.frameSize();
129
130 resize(marginW + cm.left() + cm.right() + size.width(),
131 marginH + cm.top() + cm.bottom() + size.height());
132
133 if(frameGeometry().top() > m_pMain->frameGeometry().height() >> 1)
134 move(frameGeometry().left(),
135 m_pMain->frameGeometry().height() - frameGeometry().height());
136 return 0;
137}

139void CFrmFullScreenToolBar::slotTimeOut()
140{
141 int area = 5;
142 if(m_isHide) return;
143
144 if(m_pNail->isChecked()) return;
145
146 m_isHide = true;
147 m_Timer.stop();
148 m_ToolBar.hide();
149 resize(width(), area);
150
151 int h = m_pMain->frameGeometry().height() >> 1;
152 if(frameGeometry().top() < h)
153 move(frameGeometry().left(), 0);
154 else
155 move(frameGeometry().left(), m_pMain->frameGeometry().height() - area);
156}
