48int CClient::LoadPlugins()
49{
50 foreach (QObject *plugin, QPluginLoader::staticInstances())
51 {
52 CPluginClient* p = qobject_cast<CPluginClient*>(plugin);
53 if(p)
54 {
55 if(m_Plugins.find(p->Id()) == m_Plugins.end())
56 {
57 m_Plugins.insert(p->Id(), p);
58 p->InitTranslator();

71#if defined (Q_OS_WINDOWS)

73#else

79int CClient::FindPlugins(QDir dir, QStringList filters)
80{
81 QString fileName;
82 if(filters.isEmpty())
83 {
84#if defined (Q_OS_WINDOWS)
85 filters << "*PluginClient*.dll";
86#else
87 filters << "*PluginClient*.so";
88#endif
89 }
90
91 QString szCurrentPath = QDir::currentPath();
92 QStringList files = dir.entryList(filters, QDir::Files | QDir::NoDotAndDotDot);
93 if(!files.isEmpty())
94 {
95 //This method is invalid
96 //QCoreApplication::addLibraryPath(QDir::cleanPath(dir.absolutePath()));
97
98 QDir::setCurrent(QDir::cleanPath(QDir::cleanPath(dir.absolutePath())));
99
100 // This method is valid
101//#if defined(Q_OS_WINDOWS)
102// QString szPath = QString::fromLocal8Bit(qgetenv("PATH"));
103// szPath += ";";
104// szPath += QDir::cleanPath(QDir::cleanPath(dir.absolutePath()));
105// qputenv("PATH", szPath.toLatin1());
106//#endif
107 }
108
109 foreach (fileName, files) {
110 QString szPlugins = dir.absoluteFilePath(fileName);
111 QPluginLoader loader(szPlugins);
112 QObject *plugin = loader.instance();
113 if (plugin) {
114 CPluginClient* p = qobject_cast<CPluginClient*>(plugin);
115 if(p)
116 {
117 if(m_Plugins.find(p->Id()) == m_Plugins.end())
118 {
119 m_Plugins.insert(p->Id(), p);
120 p->InitTranslator();
121 }
122 else
123 qCritical(Client) << "The plugin [" << p->Name() << "] is exist.";
124 }
125 }else{
126 QString szMsg;
127 szMsg = "load plugin error: " + loader.errorString();
128 qCritical(Client) << szMsg;
129 }
130
131 foreach (fileName, dir.entryList(QDir::Dirs | QDir::NoDotAndDotDot)) {
132 QDir pluginDir = dir;
133 if(pluginDir.cd(fileName))
134 FindPlugins(pluginDir, filters);
135 }
136 }
137
138 QDir::setCurrent(szCurrentPath);
139
140 return 0;
141}

155CConnecter* CClient::LoadConnecter(const QString &szFile)
156{
157 CConnecter* pConnecter = nullptr;
158 if(szFile.isEmpty()) return nullptr;
159
160 QSettings set(szFile, QSettings::IniFormat);
161 m_FileVersion = set.value("Manage/FileVersion", m_FileVersion).toInt();
162 QString id = set.value("Plugin/ID").toString();
163 QString protocol = set.value("Plugin/Protocol").toString();
164 QString name = set.value("Plugin/Name").toString();
165
166 qInfo(Client) << "protocol:" << protocol << "name:" << name;
167 pConnecter = CreateConnecter(id);
168 if(pConnecter)
169 pConnecter->Load(szFile);
170 else
171 qCritical(Client) << "Don't create connecter:" << protocol;
172
173 return pConnecter;
